const hojas = document.querySelectorAll('.hoja'); const grupo1 = [0, 1, 2]; const grupo2 = [3, 4, 5, 6]; function mostrarGrupo(indices, delay=8000) { indices.forEach((i, index) => { setTimeout(() => { hojas[i].classList.add('mostrar'); }, index * delay); }); } function ocultarGrupo(indices, delay=8000) { indices.forEach((i, index) => { setTimeout(() => { hojas[i].classList.remove('mostrar'); }, index * delay); }); } // Ajustar tiempo según pantalla let tiempo = 8000; if(window.innerWidth <= 480) tiempo = 4000; // móviles más rápidos else if(window.innerWidth <= 768) tiempo = 6000; // tablets setTimeout(() => { mostrarGrupo(grupo1, tiempo); setTimeout(() => { ocultarGrupo(grupo1, tiempo); setTimeout(() => { mostrarGrupo(grupo2, tiempo); setTimeout(mostrarBotonSiguiente, tiempo*4); }, 3000); }, tiempo * grupo1.length); }, 2000); function mostrarBotonSiguiente() { document.getElementById('btnSiguiente').classList.add('mostrar'); }
